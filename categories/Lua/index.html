用lua实现一个游戏彩蛋

感觉Google很爱整这种小彩蛋，主页上经常会有，android系统设置里-->关于手机-->Android版本狂点上几下也有个小彩蛋。小时候玩过魂斗罗吧，上上下下左左右右ABAB，熟悉吧！初学lua，写了个玩玩，只是个初步的思路。

local STATE_ONE = "STATE_ONE"
local STATE_TWO = "STATE_TWO"
local STATE_SUCCESS = "STATE_SUCCESS"
local STATE_EXIT = "STATE_EXIT"

local STATE_JUMP_TABLE = {}
local STATE_DO_FUNC_TABLE ={}

local curState = STATE_ONE

func1 = function() print'do state_one func' end
func2 = function() print'do state_two func' end
func3 = function() print' success' os.exit() end

dofunc = function(state)
    STATE_DO_FUNC_TABLE[1][state]()
end

insert_stateJumpTable = function(jumptable)
    table.insert(STATE_JUMP_TABLE,jumptable)
end

set_state_Func = function(functable)
    table.insert(STATE_DO_FUNC_TABLE,functable)
end

update = function()
    for i,v in ipairs(STATE_JUMP_TABLE[1]) do
        if v[1] == curState then
            dofunc(curState)
            curState = v[2]
            return
        end
    end
end

insert_stateJumpTable( { {STATE_ONE , STATE_TWO },{STATE_TWO , STATE_SUCCESS },{STATE_SUCCESS ,STATE_EXIT  }  })

set_state_Func(
{
    STATE_ONE=func1,
    STATE_TWO=func2,
    STATE_SUCCESS = func3,
}
)

while 1 do
    update()
end

local function parseName(str)
    local _begin, _end, cls = assert(str:find('%s*([a-zA-Z][a-zA-Z0-9_]*)%s*%:?'))
    if not str:find(':', _end) then
        return cls, {}
    end
    local bases = {}
    while true do
        local base
        _begin, _end, base = str:find('%s*([a-zA-Z][a-zA-Z0-9_]*)%s*%,?', _end + 1)
        if base then
            table.insert(bases, base)
        else
            break
        end
    end
    return cls, bases
end

local function create(t, ...)
    local o = {}
    if t.__init__ then
        t.__init__(o, ...)
    end
    return setmetatable(o, {__index = t, __class__ = t})
end

function class(name)
    local env = getfenv(2)
    local clsName, bases = parseName(name)
    for i, v in ipairs(bases) do
        bases[i] = env[v]   --Replace string name with class table
    end

    return function (t)
        local meta = {__call = create, __bases__ = bases}
        meta.__index = function(nouse, key)
            for _, cls in ipairs(meta.__bases__) do
                local val = cls[key]    --Do a recursive search on each cls
                if val ~= nil then
                    return val
                end
            end
        end
        -- t.class = t  --指向类，可以用于添加类成员
        t.className = clsName --类名string
        -- t.super = bases[1] --多重继承的第一个父类
        env[clsName] = setmetatable(t, meta)
    end
end
